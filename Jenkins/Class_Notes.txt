Declarative Pipeline :
	
pipeline {
    agent any

    stages {
        stage('SCM_Checkout') {
            steps {
                echo 'Perform SCM Checkout'
				
            }
        }
        stage('Java Maven Build') {
            steps {
                echo 'Java App build'
				
            }
        }
        stage('Deploy to QA Environment') {
            steps {
                echo 'Deploy the Artifacts to QA Server'
				
            }
        }
    }
}
